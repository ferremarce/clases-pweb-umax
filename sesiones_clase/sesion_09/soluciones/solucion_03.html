Código de la Solución

// Ejercicio 3: Calculadora con Validación let continuar = "s"; // Bucle principal while (continuar === "s") { // Mostrar menú let opcion = prompt("Calculadora - Elige operación:\n" + "1 = Sumar\n" + "2 = Restar\n" + "3 = Multiplicar\n" + "4 = Dividir\n" + "5 = Salir"); // Verificar si quiere salir if (opcion === "5") { alert("¡Gracias por usar la calculadora!"); continuar = "n"; continue; } // Validar opción if (opcion !== "1" && opcion !== "2" && opcion !== "3" && opcion !== "4") { alert("Opción inválida. Intenta de nuevo."); continue; } // Pedir números let num1 = Number(prompt("Primer número:")); let num2 = Number(prompt("Segundo número:")); // Validar que sean números if (isNaN(num1) || isNaN(num2)) { alert("Error: Debes ingresar números válidos."); continue; } // Realizar operación let resultado; let operacion = ""; switch (opcion) { case "1": resultado = num1 + num2; operacion = "Suma"; break; case "2": resultado = num1 - num2; operacion = "Resta"; break; case "3": resultado = num1 * num2; operacion = "Multiplicación"; break; case "4": if (num2 === 0) { alert("Error: No se puede dividir por cero."); continue; } resultado = num1 / num2; operacion = "División"; break; } // Mostrar resultado alert(operacion + ": " + resultado); // Preguntar si continuar continuar = prompt("¿Deseas continuar? (s/n)").toLowerCase(); }
💡 Nota: Esta es UNA posible solución. Se usa isNaN() para validar números, switch para las operaciones, y toLowerCase() para manejar tanto "S" como "s". El bucle while permite múltiples operaciones.

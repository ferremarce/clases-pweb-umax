// Resaltar página activa en el navbar
document.addEventListener("DOMContentLoaded", () => {
    let currentPage = window.location.pathname.split("/").pop();

    document.querySelectorAll(".nav-link").forEach(link => {
        if (link.getAttribute("href") === currentPage) {
            link.classList.add("active");
        }
    });
});

localStorage en el Proyecto

// Guardar preferencias del usuario
function guardarPreferencia(clave, valor) {
    localStorage.setItem(clave, JSON.stringify(valor));
}

// Recuperar preferencias
function obtenerPreferencia(clave) {
    let dato = localStorage.getItem(clave);
    return dato ? JSON.parse(dato) : null;
}

// Ejemplo: tema oscuro
let temaOscuro = obtenerPreferencia("temaOscuro");
if (temaOscuro) {
    document.body.classList.add("modo-oscuro");
}

// Alternar tema
document.getElementById("btnTema").addEventListener("click", () => {
    document.body.classList.toggle("modo-oscuro");
    let activo = document.body.classList.contains("modo-oscuro");
    guardarPreferencia("temaOscuro", activo);
});

Validación de Formularios

// Validación personalizada
let form = document.getElementById("formContacto");

form.addEventListener("submit", (e) => {
    e.preventDefault();

    let nombre = document.getElementById("nombre").value.trim();
    let email = document.getElementById("email").value.trim();
    let errores = [];

    if (!nombre) {
        errores.push("El nombre es obligatorio");
    }

    let regexEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!regexEmail.test(email)) {
        errores.push("Email inválido");
    }

    if (errores.length > 0) {
        alert(errores.join("\n"));
    } else {
        // Guardar en localStorage
        let contactos = JSON.parse(localStorage.getItem("contactos")) || [];
        contactos.push({ nombre, email, fecha: new Date() });
        localStorage.setItem("contactos", JSON.stringify(contactos));
        alert("Mensaje enviado correctamente");
        form.reset();
    }
});

Publicación con GitHub Pages

GitHub Pages permite publicar sitios web estáticos gratuitamente.


  • Ideal para proyectos HTML/CSS/JS
  • Se actualiza automáticamente al hacer push
  • URL: https://usuario.github.io/mi-proyecto
  • Soporta dominio personalizado

Configurar GitHub Pages

  1. Ir al repositorio en GitHub
  2. Settings → Pages
  3. Source: "Deploy from a branch"
  4. Branch: main / (root)
  5. Click "Save"
  6. Esperar 1-2 minutos
  7. Visitar: https://usuario.github.io/mi-proyecto
Importante: El archivo index.html debe estar en la raíz del repositorio. Las rutas deben ser relativas.

Verificar el Sitio Publicado

  • Probar todas las páginas y enlaces
  • Verificar que imágenes y CSS se carguen
  • Comprobar funcionalidad en móvil y desktop
  • Corregir rutas absolutas que no funcionen
  • Actualizar el repositorio con git push
Recuerda: Usar rutas relativas para todos los recursos (../css/style.css en lugar de /css/style.css).

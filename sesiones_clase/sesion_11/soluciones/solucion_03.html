Código de la Solución

<!-- HTML --> <div class="todo-app"> <h2>Lista de Tareas</h2> <div class="input-group"> <input id="nuevaTarea" type="text" placeholder="Nueva tarea..."> <button id="btnAgregar">Agregar</button> </div> <ul id="listaTareas"></ul> </div> <!-- CSS --> .todo-app { max-width: 500px; margin: 0 auto; } .input-group { display: flex; gap: 10px; margin-bottom: 20px; } #nuevaTarea { flex: 1; padding: 10px; } #btnAgregar { background: #8B0000; color: white;} #listaTareas { list-style: none; padding: 0; } .tarea { display: flex; justify-content: space-between; align-items: center; padding: 10px; border-bottom: 1px solid #ddd; } .tarea.completada .texto { text-decoration: line-through; color: #888; } .btnEliminar { background: #dc3545; color: white; border: none; padding: 5px 10px; border-radius: 4px; cursor: pointer; } <!-- JavaScript --> <script> // Función agregar tarea function agregarTarea() { let input = document.getElementById("nuevaTarea"); let texto = input.value.trim(); if (texto === "") { alert("Escribe una tarea"); return; } // Crear elemento li let li = document.createElement("li"); li.className = "tarea"; // Span para el texto let span = document.createElement("span"); span.className = "texto"; span.textContent = texto; span.addEventListener("click", () => marcarCompletada(li)); // Botón eliminar let btn = document.createElement("button"); btn.className = "btnEliminar"; btn.textContent = "X"; btn.addEventListener("click", () => eliminarTarea(li)); // Agregar elementos al li li.appendChild(span); li.appendChild(btn); // Agregar a la lista document.getElementById("listaTareas").appendChild(li); input.value = ""; input.focus(); } // Función marcar completada function marcarCompletada(elemento) { elemento.classList.toggle("completada"); } // Función eliminar tarea function eliminarTarea(elemento) { elemento.remove(); } // Evento click del botón agregar document.getElementById("btnAgregar").addEventListener("click", agregarTarea); // Evento Enter en el input document.getElementById("nuevaTarea").addEventListener("keypress", (e) => { if (e.key === "Enter") agregarTarea(); }); </script>
💡 Nota: Se usan funciones separadas para cada acción. createElement crea nuevos elementos, appendChild los agrega al DOM, y classList.toggle alterna el estado de completada.

Código de la Solución

<!-- JavaScript --> <script> // Funciones aritméticas (arrow functions) const sumar = (a, b) => a + b; const restar = (a, b) => a - b; const multiplicar = (a, b) => a * b; // Función tradicional para validar división function dividir(a, b) { if (b === 0) return "Error: división por cero"; return a / b; } // Función que recibe la operación como callback function calcular(operacion) { let n1 = Number(prompt("Ingresa el primer número:")); let n2 = Number(prompt("Ingresa el segundo número:")); let resultado = operacion(n1, n2); alert("Resultado: " + resultado); } // Bucle principal con menú let salir = false; while (!salir) { let opcion = prompt( "--- CALCULADORA ---\n" + "1. Sumar\n2. Restar\n3. Multiplicar\n4. Dividir\n5. Salir\n\nElige una opción:" ); switch (opcion) { case "1": calcular(sumar); break; case "2": calcular(restar); break; case "3": calcular(multiplicar); break; case "4": calcular(dividir); break; case "5": salir = true; alert("¡Hasta luego!"); break; default: alert("Opción no válida. Intenta de nuevo."); } } </script>
Nota: Las operaciones sumar, restar y multiplicar se declaran como arrow functions. La función calcular() recibe la operación como callback, lo que evita repetir lógica de entrada/salida. La división usa función tradicional para manejar el caso especial con retorno temprano.

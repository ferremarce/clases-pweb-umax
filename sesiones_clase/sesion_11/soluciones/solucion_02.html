Código de la Solución

<!-- JavaScript --> <script> // Variables globales let tareas = []; let contadorId = 1; // Funciones del gestor const agregarTarea = (tareas, nombre) => { if (!nombre.trim()) { alert("El nombre no puede estar vacío."); return tareas; } tareas.push({ id: contadorId++, nombre, completada: false }); alert("Tarea agregada correctamente."); return tareas; }; // Recibe un callback para procesar cada tarea const listarTareas = (tareas, callback) => { if (tareas.length === 0) { alert("No hay tareas registradas."); return; } let mensaje = "--- LISTA DE TAREAS ---\n"; tareas.forEach(t => { mensaje += callback(t) + "\n"; }); alert(mensaje); }; // Callback para formatear una tarea const formatearTarea = (t) => { let estado = t.completada ? "Hecho" : "Pendiente"; return `#${t.id} - ${t.nombre} [${estado}]`; }; // Callback de notificación const notificar = (mensaje) => { alert("[NOTIFICACIÓN] " + mensaje); }; // Marcar tarea como completada con callback const completarTarea = (tareas, indice, callback) => { if (indice < 0 || indice >= tareas.length) { alert("Índice no válido."); return; } if (tareas[indice].completada) { alert("Esa tarea ya está completada."); return; } tareas[indice].completada = true; callback(`Tarea "${tareas[indice].nombre}" completada.`); }; // Mostrar resumen con arrow function const mostrarResumen = (tareas) => { let total = tareas.length; let completadas = tareas.filter(t => t.completada).length; let pendientes = total - completadas; alert( "--- RESUMEN ---\n" + `Total: ${total}\n` + `Completadas: ${completadas}\n` + `Pendientes: ${pendientes}` ); }; // Bucle principal let salir = false; while (!salir) { let opcion = prompt( "--- GESTOR DE TAREAS v3 ---\n" + "1. Agregar tarea\n2. Listar tareas\n3. Completar tarea\n4. Mostrar resumen\n5. Salir\n\nElige una opción:" ); switch (opcion) { case "1": { let nombre = prompt("Nombre de la tarea:"); tareas = agregarTarea(tareas, nombre); break; } case "2": listarTareas(tareas, formatearTarea); break; case "3": { let indice = Number(prompt("Índice de la tarea a completar (0-based):")); completarTarea(tareas, indice, notificar); break; } case "4": mostrarResumen(tareas); break; case "5": salir = true; alert("¡Hasta luego!"); break; default: alert("Opción no válida."); } } </script>
Nota: El Gestor de Tareas v3 refactoriza el código de la sesión 10 usando funciones. agregarTarea, formatearTarea y notificar son arrow functions. listarTareas recibe un callback para procesar cada tarea. completarTarea usa callback de notificación. El resumen usa filter() para contar tareas completadas.

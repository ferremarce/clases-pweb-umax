Código de la Solución

<!-- HTML --> <div class="galeria"> <img id="imagenGrande" src="https://via.placeholder.com/600x400/8B0000/ffffff?text=Imagen+1" alt="Imagen principal"> <div class="miniaturas"> <img class="miniatura" src="https://via.placeholder.com/150x100/8B0000/ffffff?text=1" data="https://via.placeholder.com/600x400/8B0000/ffffff?text=Imagen+1"> <img class="miniatura" src="https://via.placeholder.com/150x100/DC143C/ffffff?text=2" data="https://via.placeholder.com/600x400/DC143C/ffffff?text=Imagen+2"> <img class="miniatura" src="https://via.placeholder.com/150x100/333333/ffffff?text=3" data="https://via.placeholder.com/600x400/333333/ffffff?text=Imagen+3"> <img class="miniatura" src="https://via.placeholder.com/150x100/666666/ffffff?text=4" data="https://via.placeholder.com/600x400/666666/ffffff?text=Imagen+4"> </div> </div> <!-- CSS --> .galeria { text-align: center; } #imagenGrande { max-width: 100%; max-height: 400px; border-radius: 8px; } .miniaturas { margin-top: 15px; } .miniatura { width: 100px; margin: 5px; cursor: pointer; border: 3px solid transparent; transition: transform 0.3s, border-color 0.3s; border-radius: 4px; } .miniatura:hover, .miniatura.activa { border-color: #8B0000; transform: scale(1.05); } <!-- JavaScript --> <script> // Obtener elementos let imagenGrande = document.getElementById("imagenGrande"); let miniaturas = document.querySelectorAll(".miniatura"); // Función para cambiar imagen grande function cambiarImagen(e) { // Obtener el src de la miniatura clickeada let nuevaSrc = e.target.dataset.src || e.target.src; // Cambiar imagen grande imagenGrande.src = nuevaSrc.replace("150x100", "600x400"); // Quitar clase activa de todas miniaturas.forEach(min => min.classList.remove("activa")); // Agregar clase activa a la clickeada e.target.classList.add("activa"); } // Agregar eventos click miniaturas.forEach(min => { min.addEventListener("click", cambiarImagen); // Efecto mouseover/mouseout min.addEventListener("mouseover", (e) => { e.target.style.transform = "scale(1.1)"; }); min.addEventListener("mouseout", (e) => { e.target.style.transform = "scale(1)"; }); }); </script>
💡 Nota: Se usa querySelectorAll para obtener todas las miniaturas y forEach para agregar eventos. El atributo data-src guarda la URL de la imagen grande.

Código de la Solución

// Ejercicio 2: Juego de Adivinanza // Generar número aleatorio entre 1 y 100 const numeroSecreto = Math.floor(Math.random() * 100) + 1; let intentos = 5; let adivinado = false; alert("¡Juego de Adivinanza!\n" + "Adivina el número entre 1 y 100.\n" + "Tienes 5 intentos."); // Bucle principal while (intentos > 0 && !adivinado) { // Pedir número let intento = Number(prompt("Adivina el número (1-100).\n" + "Intentos restantes: " + intentos)); // Validar if (isNaN(intento) || intento < 1 || intento > 100) { alert("Número inválido. Debe ser entre 1 y 100."); continue; } // Comparar if (intento === numeroSecreto) { adivinado = true; alert("¡Felicidades! ¡Adivinaste!\n" + "El número era " + numeroSecreto); } else if (intento < numeroSecreto) { alert("El número es MAYOR. Intenta de nuevo."); intentos--; } else { alert("El número es MENOR. Intenta de nuevo."); intentos--; } } // Si perdido if (!adivinado) { alert("¡Perdiste! Se acabaron los intentos.\n" + "El número era: " + numeroSecreto); }
💡 Nota: Esta es UNA posible solución. Se usa Math.random() para el número aleatorio, un bucle while con contador de intentos, y comparación con if/else para dar pistas.

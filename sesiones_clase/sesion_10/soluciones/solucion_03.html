// Ejercicio 3: Simulación de Notas // Array de notas de estudiantes const notas = [75, 92, 45, 88, 70]; const nombres = ["Ana", "Juan", "Pedro", "María", "Luis"]; let suma = 0; let mensaje = "Resultados:\n\n"; // Iterar con for for (let i = 0; i < notas.length; i++) { let nota = notas[i]; let nombre = nombres[i]; let calificacion, descripcion; // Determinar calificación con switch switch (nota) { // 90-100: A case 90: case 91: case 92: case 93: case 94: case 95: case 96: case 97: case 98: case 99: case 100: calificacion = "A"; descripcion = "Excelente"; break; // 80-89: B case 80: case 81: case 82: case 83: case 84: case 85: case 86: case 87: case 88: case 89: calificacion = "B"; descripcion = "Muy Bueno"; break; // 70-79: C case 70: case 71: case 72: case 73: case 74: case 75: case 76: case 77: case 78: case 79: calificacion = "C"; descripcion = "Bueno"; break; // 60-69: D case 60: case 61: case 62: case 63: case 64: case 65: case 66: case 67: case 68: case 69: calificacion = "D"; descripcion = "Suficiente"; break; // 0-59: F default: calificacion = "F"; descripcion = "Reprobado"; } // Agregar al mensaje mensaje += nombre + ": Nota " + nota + " = " + calificacion + " (" + descripcion + ")\n"; // Sumar para promedio suma += nota; } // Calcular promedio let promedio = suma / notas.length; // Mostrar resultados alert(mensaje + "\nPromedio general: " + promedio.toFixed(2));
💡 Nota: Esta es UNA posible solución. Se usa for para iterar y switch con rangos para determinar la calificación. El promedio se calcula sumando todas las notas y dividiendo por la cantidad.

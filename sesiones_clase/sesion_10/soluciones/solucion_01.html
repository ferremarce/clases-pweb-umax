Código de la Solución

// Ejercicio 1: Calculadora con Menú // Mostrar menú let opcion = prompt("Calculadora - Elige operación:\n" + "1 = Sumar\n" + "2 = Restar\n" + "3 = Multiplicar\n" + "4 = Mayor\n" + "5 = Par/Impar"); // Procesar según opción if (opcion === "1") { // Sumar let n1 = Number(prompt("Primer número:")); let n2 = Number(prompt("Segundo número:")); alert("Resultado: " + (n1 + n2)); } else if (opcion === "2") { // Restar let n1 = Number(prompt("Primer número:")); let n2 = Number(prompt("Segundo número:")); alert("Resultado: " + (n1 - n2)); } else if (opcion === "3") { // Multiplicar let n1 = Number(prompt("Primer número:")); let n2 = Number(prompt("Segundo número:")); alert("Resultado: " + (n1 * n2)); } else if (opcion === "4") { // Mayor let n1 = Number(prompt("Primer número:")); let n2 = Number(prompt("Segundo número:")); let mayor = n1 > n2 ? n1 : n2; alert("El mayor es: " + mayor); } else if (opcion === "5") { // Par/Impar let n1 = Number(prompt("Número:")); let resultado = n1 % 2 === 0 ? "Par" : "Impar"; alert(n1 + " es " + resultado); } else { alert("Opción inválida"); }
💡 Nota: Esta es UNA posible solución. Se usa if/else encadenado para cada opción. También podría usar switch o una función para cada operación.

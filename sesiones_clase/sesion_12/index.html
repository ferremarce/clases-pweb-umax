Crear Elementos: createElement

// Crear un nuevo elemento
let nuevoDiv = document.createElement("div");

// Agregar contenido
nuevoDiv.textContent = "Hola mundo";

// Agregar clase
nuevoDiv.className = "nuevo-elemento";

// Establecer atributos
nuevoDiv.setAttribute("id", "miId");
nuevoDiv.setAttribute("data-info", "ejemplo");

Agregar Elementos al DOM

// Crear elemento
let nuevoParrafo = document.createElement("p");
nuevoParrafo.textContent = "Nuevo párrafo";

// appendChild - agregar al final
contenedor.appendChild(nuevoParrafo);

// insertBefore - agregar antes de otro
contenedor.insertBefore(nuevoParrafo, primerHijo);

// prepend - agregar al inicio
contenedor.prepend(nuevoParrafo);

// after - agregar después
elemento.after(nuevoParrafo);

Eliminar Elementos

// remove() - eliminar directamente
elemento.remove();

// removeChild() - eliminar hijo
padre.removeChild(hijo);

// Reemplazar elemento
padre.replaceChild(nuevoElemento, elementoAntiguo);

// Vaciar contenido
elemento.innerHTML = "";
elemento.textContent = "";

Modificar Atributos

// getAttribute() - obtener valor
let valor = elemento.getAttribute("class");

// setAttribute() - establecer valor
elemento.setAttribute("class", "nueva-clase");

// removeAttribute() - eliminar atributo
elemento.removeAttribute("disabled");

// Booleanos (checked, disabled, etc.)
elemento.checked = true;
elemento.disabled = false;

localStorage: Introducción

localStorage permite guardar datos en el navegador de forma persistente.


  • Persiste entre sesiones
  • Hasta 5MB de datos
  • Solo guarda strings
  • Accesible desde cualquier pestaña
💡 Nota: Los datos persisten incluso si cierras el navegador.

localStorage: Guardar Datos

// Guardar un string
localStorage.setItem("nombre", "Juan");

// Guardar número (se guarda como string)
localStorage.setItem("edad", 30);

// Guardar con clave simple
localStorage["email"] = "juan@email.com";

// Obtener valor
let nombre = localStorage.getItem("nombre");

// Verificar si existe
if (localStorage.getItem("nombre")) {
    // existe
}

localStorage: Recuperar y Eliminar

// Obtener valor
let nombre = localStorage.getItem("nombre");

// Eliminar un ítem
localStorage.removeItem("nombre");

// Eliminar todos los ítems
localStorage.clear();

// Cantidad de ítems
let cantidad = localStorage.length;

JSON en localStorage

localStorage solo guarda strings. Usa JSON para objetos y arrays.


// Guardar objeto (convertir a JSON)
let usuario = { nombre: "Juan", edad: 30 };
localStorage.setItem("usuario", JSON.stringify(usuario));

// Recuperar objeto (convertir a objeto)
let usuarioGuardado = JSON.parse(localStorage.getItem("usuario"));

// Guardar array
let tareas = ["Estudiar", "Trabajar"];
localStorage.setItem("tareas", JSON.stringify(tareas));
⚠️ Importante: Siempre usa JSON.stringify() al guardar y JSON.parse() al recuperar.

sessionStorage

sessionStorage es similar a localStorage pero:


  • Se elimina al cerrar la pestaña
  • Solo accesible en esa pestaña
  • Mismos métodos: setItem, getItem, removeItem

// Guarda datos solo durante la sesión
sessionStorage.setItem("temporal", "dato");

Proyecto: To-Do List con Persistencia

// Estructura de datos en localStorage
let tareas = [
    { texto: "Estudiar JS", completada: false },
    { texto: "Hacer ejercicio", completada: true }
];

// Guardar en localStorage
localStorage.setItem("tareas", JSON.stringify(tareas));

// Recuperar al cargar
let tareasGuardadas = JSON.parse(localStorage.getItem("tareas")) || [];

// Agregar nueva tarea
function agregarTarea(texto) {
    tareas.push({ texto: texto, completada: false });
    localStorage.setItem("tareas", JSON.stringify(tareas));
}

// Eliminar tarea
function eliminarTarea(indice) {
    tareas.splice(indice, 1);
    localStorage.setItem("tareas", JSON.stringify(tareas));
}

Validación de Formularios

// Validar EMAIL con expresión regular
function validarEmail(email) {
    let regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return regex.test(email);
}

// Validar required
function validarRequerido(valor) {
    return valor.trim() !== "";
}

// Validar longitud
function validarLongitud(valor, min, max) {
    return valor.length >= min && valor.length <= max;
}

// Usar en submit
form.addEventListener("submit", (e) => {
    if (!validarEmail(emailInput.value)) {
        e.preventDefault();
        alert("Email inválido");
    }
});

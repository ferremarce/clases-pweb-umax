El DOM (Document Object Model)

El DOM es la representación en memoria del documento HTML como objetos JavaScript.


  • document: objeto principal
  • Cada etiqueta HTML es un elemento
  • JavaScript puede modificarlo
  • Árbol jerárquico de nodos
Pruébalo: Abre la consola y escribe: document.body

Seleccionar Elementos

// Por ID (más rápido)
let elemento = document.getElementById("miId");

// Por selector CSS (más flexible)
let elemento = document.querySelector(".miClase");
let elementos = document.querySelectorAll(".miClase");

// Por nombre de etiqueta
let parrafos = document.getElementsByTagName("p");
Importante: Si el elemento no existe, getElementById retorna null. Siempre verificar antes de usar.

Eventos del DOM

Los eventos son acciones que ocurren en la página.


Evento Descripción
click Cuando se hace click
mouseover Cuando el mouse entra
mouseout Cuando el mouse sale
submit Al enviar formulario
input Al escribir en input
keyup Cuando se libera una tecla

addEventListener

// Seleccionar elemento
let boton = document.getElementById("miBoton");

// Agregar evento con función anónima
boton.addEventListener("click", function() {
    alert("Hiciste click");
});

// Con arrow function
boton.addEventListener("click", () => {
    alert("Hiciste click");
});

// Con función nombrada (reutilizable)
function manejarClick() {
    alert("Hiciste click");
}
boton.addEventListener("click", manejarClick);

Modificar Contenido

// Cambiar texto (seguro, no interpreta HTML)
elemento.textContent = "Nuevo texto";

// Cambiar HTML interno (interpreta etiquetas)
elemento.innerHTML = "<strong>Negrita</strong>";

// Obtener valor de input
let valor = inputElement.value;

// Establecer valor
inputElement.value = "Nuevo valor";
Seguridad: Usar textContent para texto plano. innerHTML puede ser peligroso con datos del usuario.

Modificar Estilos

// Cambiar estilos inline
elemento.style.color = "red";
elemento.style.backgroundColor = "yellow";
elemento.style.display = "none";

// camelCase para propiedades CSS con guión
// background-color → backgroundColor
// font-size → fontSize

// Manipular clases
elemento.classList.add("activa");
elemento.classList.remove("oculta");
elemento.classList.toggle("visible");

Crear Elementos: createElement

// Crear un nuevo elemento
let nuevoDiv = document.createElement("div");

// Agregar contenido
nuevoDiv.textContent = "Hola mundo";

// Agregar clase
nuevoDiv.className = "nuevo-elemento";

// Establecer atributos
nuevoDiv.setAttribute("id", "miId");
nuevoDiv.setAttribute("data-info", "ejemplo");

Agregar Elementos al DOM

let nuevoParrafo = document.createElement("p");
nuevoParrafo.textContent = "Nuevo párrafo";

// Agregar al final del contenedor
contenedor.appendChild(nuevoParrafo);

// Agregar al inicio
contenedor.prepend(nuevoParrafo);

// Insertar antes de otro elemento
contenedor.insertBefore(nuevoParrafo, referencia);

// Insertar después de un elemento
elemento.after(nuevoParrafo);

Eliminar Elementos

// Eliminar directamente
elemento.remove();

// Eliminar hijo del padre
padre.removeChild(hijo);

// Reemplazar elemento
padre.replaceChild(nuevoElemento, elementoAntiguo);

// Vaciar contenido
elemento.innerHTML = "";
elemento.textContent = "";

Modificar Atributos

// Obtener valor
let valor = elemento.getAttribute("class");

// Establecer valor
elemento.setAttribute("class", "nueva-clase");

// Eliminar atributo
elemento.removeAttribute("disabled");

// Atributos booleanos
elemento.checked = true;
elemento.disabled = false;

// Atributos data-*
elemento.dataset.info = "valor";

localStorage: Introducción

localStorage permite guardar datos en el navegador de forma persistente.


  • Persiste entre sesiones (al cerrar y abrir el navegador)
  • Hasta 5MB de datos por dominio
  • Solo guarda strings
  • Accesible desde cualquier pestaña del mismo dominio

localStorage: Guardar y Recuperar

// Guardar un valor
localStorage.setItem("nombre", "Juan");

// Guardar número (se convierte a string)
localStorage.setItem("edad", "30");

// Obtener valor
let nombre = localStorage.getItem("nombre");

// Verificar si existe
if (localStorage.getItem("nombre")) {
    console.log("El usuario es: " + nombre);
}

// Eliminar un ítem
localStorage.removeItem("nombre");

// Limpiar todo
localStorage.clear();

JSON en localStorage

localStorage solo guarda strings. Usa JSON para objetos y arrays.


// Guardar objeto
let usuario = { nombre: "Juan", edad: 30 };
localStorage.setItem("usuario", JSON.stringify(usuario));

// Recuperar objeto
let usuarioGuardado = JSON.parse(
    localStorage.getItem("usuario")
);

// Guardar array de tareas
let tareas = ["Estudiar", "Trabajar"];
localStorage.setItem("tareas", JSON.stringify(tareas));
Importante: Siempre usa JSON.stringify() al guardar y JSON.parse() al recuperar objetos o arrays.

sessionStorage

sessionStorage es similar a localStorage pero:


  • Los datos se eliminan al cerrar la pestaña
  • No se comparte entre pestañas
  • Mismos métodos: setItem, getItem, removeItem, clear

// Guardar dato temporal
sessionStorage.setItem("temporal", "dato");

// Recuperar
let dato = sessionStorage.getItem("temporal");

Validación de Formularios

// Validar email con expresión regular
function validarEmail(email) {
    let regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return regex.test(email);
}

// Validar campo requerido
function validarRequerido(valor) {
    return valor.trim() !== "";
}

// Validar longitud
function validarLongitud(valor, min, max) {
    return valor.length >= min && valor.length <= max;
}

// Usar en evento submit
form.addEventListener("submit", (e) => {
    if (!validarEmail(emailInput.value)) {
        e.preventDefault();
        alert("Email inválido");
    }
});

Ejemplo: To-Do List con localStorage

let tareas = JSON.parse(localStorage.getItem("tareas")) || [];

function agregarTarea(texto) {
    tareas.push({ texto, completada: false });
    localStorage.setItem("tareas", JSON.stringify(tareas));
    renderizarTareas();
}

function eliminarTarea(indice) {
    tareas.splice(indice, 1);
    localStorage.setItem("tareas", JSON.stringify(tareas));
    renderizarTareas();
}

function toggleTarea(indice) {
    tareas[indice].completada = !tareas[indice].completada;
    localStorage.setItem("tareas", JSON.stringify(tareas));
    renderizarTareas();
}

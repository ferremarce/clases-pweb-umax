Solución: Ejercicio 2 - Gestor de Tareas v4 con DOM

Sesión 12 - JavaScript IV

Prof. Ing. Juan M. Ferreira

Código de la Solución

<!-- HTML --> <div class="gestor"> <h2>Gestor de Tareas v4</h2> <input id="tareaInput" placeholder="Nueva tarea..."> <button id="agregarBtn">Agregar</button> <ul id="listaTareas"></ul> </div> <!-- JavaScript --> <script> // Array de tareas en memoria let tareas = []; let contadorId = 1; // Seleccionar elementos del DOM let inputTarea = document.getElementById("tareaInput"); let listaTareas = document.getElementById("listaTareas"); let agregarBtn = document.getElementById("agregarBtn"); // Función para renderizar la lista desde el array function renderizarTareas() { // Limpiar lista actual listaTareas.innerHTML = ""; tareas.forEach((tarea, indice) => { // Crear elemento li let li = document.createElement("li"); li.textContent = tarea.nombre; // Si está completada, agregar clase if (tarea.completada) { li.classList.add("completada"); } // Botón Completar let btnCompletar = document.createElement("button"); btnCompletar.textContent = tarea.completada ? "Deshacer" : "Completar"; btnCompletar.addEventListener("click", () => { tareas[indice].completada = !tareas[indice].completada; renderizarTareas(); }); // Botón Eliminar let btnEliminar = document.createElement("button"); btnEliminar.textContent = "Eliminar"; btnEliminar.addEventListener("click", () => { tareas.splice(indice, 1); renderizarTareas(); }); li.appendChild(btnCompletar); li.appendChild(btnEliminar); listaTareas.appendChild(li); }); } // Agregar tarea function agregarTarea() { let texto = inputTarea.value.trim(); if (texto) { tareas.push({ id: contadorId++, nombre: texto, completada: false }); inputTarea.value = ""; renderizarTareas(); } } // Eventos agregarBtn.addEventListener("click", agregarTarea); inputTarea.addEventListener("keyup", (e) => { if (e.key === "Enter") agregarTarea(); }); // Render inicial (vacío) renderizarTareas(); </script>
Nota: El Gestor de Tareas v4 usa manipulación del DOM: createElement, appendChild, classList.add, y eventos dinámicos. Los datos viven en un array en memoria, sin localStorage. Cada tarea tiene botones Completar/Deshacer y Eliminar creados dinámicamente.

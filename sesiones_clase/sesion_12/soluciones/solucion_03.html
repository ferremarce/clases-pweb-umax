Solución: Ejercicio 3 - To-Do List Completa

Sesión 12 - JavaScript IV

Prof. Ing. Juan M. Ferreira

Código de la Solución

<!-- JavaScript --> <script> // Array de tareas [{ texto, completada }] let tareas = []; // Cargar tareas desde localStorage function cargarTareas() { let guardadas = localStorage.getItem("tareas"); if (guardadas) { tareas = JSON.parse(guardadas); } renderizar(); } // Guardar tareas en localStorage function guardarTareas() { localStorage.setItem("tareas", JSON.stringify(tareas)); } // Renderizar la lista function renderizar() { let lista = document.getElementById("listaTareas"); lista.innerHTML = ""; tareas.forEach((tarea, indice) => { let li = document.createElement("li"); li.className = tarea.completada ? "completada" : ""; // Span con texto let span = document.createElement("span"); span.textContent = tarea.texto; span.addEventListener("click", () => marcarCompletada(indice)); li.appendChild(span); // Botón eliminar let btn = document.createElement("button"); btn.textContent = "X"; btn.addEventListener("click", () => eliminarTarea(indice)); li.appendChild(btn); lista.appendChild(li); }); } // Agregar tarea function agregarTarea() { let input = document.getElementById("tareaInput"); let texto = input.value.trim(); if (texto) { tareas.push({ texto: texto, completada: false }); guardarTareas(); renderizar(); input.value = ""; } } // Marcar tarea completada function marcarCompletada(indice) { tareas[indice].completada = !tareas[indice].completada; guardarTareas(); renderizar(); } // Eliminar tarea function eliminarTarea(indice) { tareas.splice(indice, 1); guardarTareas(); renderizar(); } // Eventos document.getElementById("agregarBtn").addEventListener("click", agregarTarea); document.getElementById("tareaInput").addEventListener("keyup", (e) => { if (e.key === "Enter") agregarTarea(); }); // Iniciar cargarTareas(); </script>
💡 Nota: Se usa un array de objetos con la estructura { texto, completada }. JSON.stringify() convierte el array a string para localStorage, y JSON.parse() lo convierte de vuelta a array.

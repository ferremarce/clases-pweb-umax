Prof. Ing. Juan M. Ferreira

Código de la Solución

<!-- HTML --> <div class="app"> <h2>Lista de Tareas</h2> <input id="tareaInput" placeholder="Nueva tarea..."> <button id="agregarBtn">Agregar</button> <ul id="lista"></ul> </div> <!-- JavaScript --> <script> // Array de tareas let tareas = []; // Cargar tareas al iniciar function cargarTareas() { let guardadas = localStorage.getItem("misTareas"); if (guardadas) { tareas = JSON.parse(guardadas); } renderizar(); } // Guardar en localStorage function guardarTareas() { localStorage.setItem("misTareas", JSON.stringify(tareas)); } // Renderizar la lista function renderizar() { let lista = document.getElementById("lista"); lista.innerHTML = ""; tareas.forEach((tarea, indice) => { let li = document.createElement("li"); li.textContent = tarea; lista.appendChild(li); }); } // Agregar tarea function agregarTarea() { let input = document.getElementById("tareaInput"); let texto = input.value.trim(); if (texto) { tareas.push(texto); guardarTareas(); renderizar(); input.value = ""; } } // Evento click document.getElementById("agregarBtn").addEventListener("click", agregarTarea); // Evento Enter document.getElementById("tareaInput").addEventListener("keyup", (e) => { if (e.key === "Enter") agregarTarea(); }); // Iniciar cargarTareas(); </script>
💡 Nota: Se usa localStorage para persistir. Las tareas se guardan como JSON string y se recuperan parseando el JSON.

Código de la Solución

<!-- HTML --> <div class="calculadora"> <h2>Calculadora</h2> <input id="num1" type="number" placeholder="Número 1"> <input id="num2" type="number" placeholder="Número 2"> <div class="botones"> <button id="btnSumar">+</button> <button id="btnRestar">-</button> <button id="btnMultiplicar">×</button> <button id="btnDividir">÷</button> </div> <div id="resultado">Resultado: -</div> </div> <!-- JavaScript --> <script> // Funciones de operación (reutilizadas de sesión 11) const sumar = (a, b) => a + b; const restar = (a, b) => a - b; const multiplicar = (a, b) => a * b; function dividir(a, b) { if (b === 0) return "Error: división por cero"; return a / b; } // Seleccionar elementos del DOM let num1Input = document.getElementById("num1"); let num2Input = document.getElementById("num2"); let resultadoDiv = document.getElementById("resultado"); // Función que ejecuta una operación y muestra resultado function calcular(operacion) { let n1 = Number(num1Input.value); let n2 = Number(num2Input.value); let resultado = operacion(n1, n2); resultadoDiv.textContent = "Resultado: " + resultado; } // Agregar eventos a los botones document.getElementById("btnSumar").addEventListener("click", () => calcular(sumar)); document.getElementById("btnRestar").addEventListener("click", () => calcular(restar)); document.getElementById("btnMultiplicar").addEventListener("click", () => calcular(multiplicar)); document.getElementById("btnDividir").addEventListener("click", () => calcular(dividir)); </script>
Nota: Se reutilizan las funciones de la sesión 11 (sumar, restar, multiplicar como arrow functions, dividir con validación). La función calcular() recibe la operación como callback. Los eventos se asignan con addEventListener y el resultado se muestra con textContent.
